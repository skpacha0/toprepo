pipeline {
    agent any 
    stages {
        stage('print hello') { 
            steps {
                sh "echo 'HELLO'"
            }
        }
        stage('github repo') { 
            steps {
		sh "rm -rf toprepo"
                sh "git clone https://github.com/skpacha0/toprepo.git"
                

                
            }
        }
        stage('build artifact') { 
            steps {
		sh "mvn clean -f toprepo"
		sh "mvn install -f toprepo"
		sh "mvn package -f toprepo"
                
            }
        }
    }
}
